• readerData ( когда выдана )

• getBookInfo() - посмотреть информацию о наличии книги

( вывести в консоль readerData )

• getTheBook( client ) - получить книгу:

проверка, что книга не выдана на руки

( приватное свойство readerName должно быть пустой строкой )

если не выдана, то вызвать приватный метод giveTheBook ( client ) и вернуть bookTitle

• returnBook() - вернуть книгу:

var LibraryBook = function(title, year, author) {
    var title = title
    var year = year
    var author = author
    var readerName = ''
    var readerData = ''

    function giveTheBook(client, date = new Date()) {
      readerName = client
      readerData = date
    }

    this.getBookInfo = function() {
      var text = readerName ?
      `Книга выданна: ${readerData}` :
      `В наличии: ${title}, ${author}, ${year}`
      console.log(text)
    }

    this.getTheBook = function(client, date) {
      if(readerName) return null
      giveTheBook(client, date)
      return title
    }

    this.returnBook = function() {
      readerName = '', readerData = ''
    }
}

var book_1 = new LibraryBook('По ком звонит колокол', 1995, 'Эрнест Хемингуэй')
